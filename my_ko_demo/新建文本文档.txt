<div data-bind="text:outputNameColor()"></div>
<script>
    class vm{
        constructor(){
            this.name = ko.observable("zhangsan");
            this.color = ko.observable("#cccc");
        }
        outputNameColor(){
            return this.name() + " " +this.color();
        }
    }

   ko.applyBindings(new vm())


</script>
